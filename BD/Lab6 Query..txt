Sarcina 3: 
SELECT DISTINCT p.Nume_Profesor, p.Prenume_Profesor, d.Disciplina
FROM studenti_reusita sr

JOIN profesori p ON sr.Id_Profesor = p.Id_Profesor
JOIN discipline d ON sr.Id_Disciplina = d.Id_Disciplina

ORDER BY p.Nume_Profesor DESC, p.Prenume_Profesor DESC, d.Disciplina DESC;
